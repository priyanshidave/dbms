CREATE OR REPLACE TRIGGER TRIG1
     AFTER UPDATE ON EMP1 
     FOR EACH ROW
DECLARE 
    DIFF NUMBER(10);
BEGIN
    DIFF := :NEW.SALARY - :OLD.SALARY;
    IF DIFF > 2000 THEN 
    DBMS_OUTPUT.PUT_LINE('ERROR : NEW SALARY IS HIGHER ');
    END IF;
END


UPDATE EMP1 SET SALARY=40000 WHERE EMPNO=111;
